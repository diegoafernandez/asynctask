 57 abstract class AsyncTask

 59     /**
 60      * A numeric shared memory segment ID
 61      * @var integer
 62      * @static
 63      */
 64     private static $shmId;
 65 
 66     /**
 67      * The line number in which the object has been initialized
 68      * @var integer
 69      * @static
 70      */
 71     private $line;
 72 
 73     /**
 74      * Creates a new asynchronous task
 75      * @return void
 76      * @access public
 77      * @final
 78      */
 79     final public function __construct()
 80     {
 81         $error = "";
 82         if (version_compare(PHP_VERSION, '5.3.3', '<')) {
 83             $error .= sprintf(
 84                 "\n\e[0m\e[0;32mAsyncTask only officially supports PHP 5.3.3 and above,\e[0m",
 85                 PHP_VERSION
 86             );
 87         }
 88         if (!extension_loaded('pcntl')) {
 89             $error .= "\n\e[0m\e[0;32mAsyncTask uses the extension \"pcntl\",\e[0m";
 90         }
 91         if (!extension_loaded('posix')) {
 92             $error .= "\n\e[0m\e[0;32mAsyncTask uses the extension \"posix\",\e[0m";
 93         }
 94 
 95         if (mb_strlen($error) > 0) {
 96             throw new RuntimeException(
 97                $error . "\n\e[0m\e[0;32myou will most likely encounter problems with non-installed extensions,"
 98                       . "\n\e[0;31mupgrading is strongly recommended.\e[0m\n"
 99            );
100         }
101 
102         $line = debug_backtrace();
103         $this->line = $line[0]['line'];
104 
105         self::$shmId = shm_attach((int) (ftok(__FILE__, 'A') . $this->line));
106         shm_put_var(self::$shmId, 11511697116117115, 'PENDING');
107         shm_put_var(self::$shmId, 112112105100, getmypid());
108     }
109 
110     /**
111      * Finish create an asynchronous task
112      * @return void
113      * @access public
114      * @final
115      */
116     final public function __destruct()
117     {
118         if (
119             @shm_has_var(self::$shmId, 112112105100) &&
120             shm_get_var(self::$shmId, 112112105100) == getmypid()
121         ) {
122             shm_remove(self::$shmId);
123         }
124     }

126     /**
127      * Returns the variable with the given key
128      * @param string $key
129      * @return mixed
130      * @access public
131      * @static
132      * @final
133      */
134     final public static function getProperty($key)
135     {
136         if (
137             in_array($key, array('shmId', 'pid', 'ppid', 'status')) === false &&
138             @shm_has_var(self::$shmId, self::getUid($key))
139         ) {
140             return shm_get_var(self::$shmId, self::getUid($key));
141         } else {
142             return false;
143         }
144     }
145 
146     /**
147      * Inserts or updates a variable with the given key
148      * @param string $key
149      * @param string $value
150      * @return boolean
151      * @access public
152      * @static
153      * @final
154      */
155     final public static function setProperty($key, $value)
156     {
157         if (in_array($key, array('shmId', 'pid', 'ppid', 'status')) === false) {
158             shm_put_var(self::$shmId, self::getUid($key), $value);
159             return true;
160         } else {
161             return false;
162         }
163     }

165     /**
166      * Returns a unique integer identifier for a given key
167      * @param string $key
168      * @return integer
169      * @access private
170      * @static
171      * @final
172      */
173     final private static function getUid($key)
174     {
175         $uid = '';
176         for ($char = 0; $char < strlen($key); $char++) {
177             $uid .= ord($key[ $char ]);
178         }
179 
180         return (int) $uid;
181     }

183     /**
184      * Executes the task with the specified parameters
185      * @param mixed $parameters
186      * @return void
187      * @access public
188      * @final
189      */
190     final public function execute($parameters)
191     {
192         $pid = pcntl_fork();
193         if ($pid == -1) {
194             exit();
195         } elseif (!$pid) {
196             self::$shmId = shm_attach((int) (ftok(__FILE__, 'A') . $this->line));
197             shm_put_var(self::$shmId, 112105100, getmypid());
198             shm_put_var(self::$shmId, 11511697116117115, 'RUNNING');
199 
200             $this->onPreExecute();
201 
202             $result = $this->doInBackground($parameters);
203 
204             $this->onPostExecute($result);
205 
206             if (@shm_has_var(self::$shmId, 112105100)) {
207                 shm_put_var(self::$shmId, 112105100, null);
208                 shm_put_var(self::$shmId, 11511697116117115, 'FINISHED');
209             }
210             exit();
211         }
212     }

214     /**
215      * Attempts to cancel execution of this task
216      * @return boolean
217      * @access public
218      * @final
219      */
220     final public function cancel()
221     {
222         if (
223             @shm_has_var(self::$shmId, 112105100) &&
224             is_null(shm_get_var(self::$shmId, 112105100)) !== null
225         ) {
226             $this->onCancelled();
227             posix_kill(shm_get_var(self::$shmId, 112105100), SIGKILL);
228             shm_put_var(self::$shmId, 112105100, null);
229             shm_put_var(self::$shmId, 11511697116117115, 'CANCELED');
230             return true;
231         } else {
232             return false;
233         }
234     }

236     /**
237      * Returns the current status of this task
238      * @return string
239      * @access public
240      * @final
241      */
242     final public function getStatus()
243     {
244         return @shm_has_var(self::$shmId, 11511697116117115)
245                    ? shm_get_var(self::$shmId, 11511697116117115)
246                    : 'PENDING';
247     }
248 
249     /**
250      * Returns true if this task was cancelled before it completed normally
251      * @return boolean
252      * @access public
253      * @final
254      */
255     final public function isCancelled()
256     {
257         return @shm_has_var(self::$shmId, 11511697116117115) == 'CANCELED' ? true : false;
258     }
